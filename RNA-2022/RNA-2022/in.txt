$Пользовательские функции
function number power [number a, number b]
{
	def number i = 2;
	def number res = a;
	
	when[b == 0]
	{
		give[1];
	}
	
	$Цикл: пока i меньше либо равно n
	cycle [i <= b]
	{
		res = res * a;
		i = i + 1;
	}
	give [res];
}

$Процедура
procedure yourLogin [line login]
{
	outln["Смена логина проведена успешно!"];
	out["Ваш новый логин: "];
	outln[login];
}

	out["Результат был: "]; outln[expr];
